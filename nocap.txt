-- Menu
local ScreenGui = Instance.new("ScreenGui", game.CoreGui)
local Frame = Instance.new("Frame", ScreenGui)
local Button = Instance.new("TextButton", Frame)

Frame.Size = UDim2.new(0,200,0,120)
Frame.Position = UDim2.new(0.05,0,0.3,0)
Frame.BackgroundColor3 = Color3.fromRGB(25,25,25)
Frame.Active = true
Frame.Draggable = true

Button.Size = UDim2.new(0,180,0,50)
Button.Position = UDim2.new(0,10,0,30)
Button.Text = "Auto Gom Quái : OFF"
Button.BackgroundColor3 = Color3.fromRGB(0,170,255)

local auto = false

Button.MouseButton1Click:Connect(function()
    auto = not auto
    Button.Text = auto and "Auto Gom Quái : ON" or "Auto Gom Quái : OFF"
end)

-- Auto Gom 4 Quái
spawn(function()
while task.wait(0.3) do
    if auto then
        local enemies = {}
        
        for _,v in pairs(game.Workspace.Enemies:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") then
                if v.Humanoid.Health > 0 then
                    table.insert(enemies,v)
                end
            end
        end
        
        for i = 1, math.min(4,#enemies) do
            enemies[i].HumanoidRootPart.CFrame =
            game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame * CFrame.new(0,0,-5)
        end
    end
end
end)
